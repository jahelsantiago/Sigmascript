function f1(streak_model, alpha, blocks, treatments, return){
	x1 = alpha^0.33 / sin(len(treatments));
	i = 0;
	j = 0;
	p_result = 1;
	s_result = 0; 
	while(i < len(blocks)){
		while(j < len(treatments)){
			op1 = 28*pi;
			frac1 = streak_model.operationAvg(i,j,.)^blocks[0] - arcsin(streak_model.operationSum(.,.,.));
			frac2 = treatments[j]*streak_model.operationSum(i,.,.)^0.67 - blocks[i]*streak_model.operationSum(.,i,.)^8;
			r1 = op1 * frac1 / frac2;
			p_result = p_result *r1; 

		}
		s_result = s_result + p_result;
	}
	x2 = s_result;
	return = x1 + x2;
}

r_f1 = 0;
call f1(streak_model, alpha, blocks, treatments, r_f1);
